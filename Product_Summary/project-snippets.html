  text:       "...",      // the task title — the one field the user always sees
  bucket:     "inbox",    // which sidebar bucket this task belongs to
  done:       false,      // true when the task has been completed
  created:    Date.now(), // Unix timestamp in ms — used for sorting
  priority:   [],         // array of strings: ["Imperative", "Financial", ...]
  location:   [],         // array of strings: ["Home", "Work", ...]
  dueDate:    null,       // YYYY-MM-DD string, or null
  deferUntil: null,       // YYYY-MM-DD string; when set and in the future, task is hidden
  effort:     null,       // single string: "2 hours", "1 day", etc.
  notes:      null,       // free-text string or null
}

// ── Root project task — carries a childIds array ──
{
  ...same fields...,
  bucket:   "project",
  childIds: ["id-of-first-subtask", ...],  // ordered list of direct child IDs
  // parentId is absent (or null) — no parent means this is a root project
}

// ── Subtask — carries a parentId ──
{
  ...same fields...,
  bucket:   "next",         // subtasks live in the Next Actions bucket
  parentId: "id-of-parent", // points to the project (or another next action) above it
  // childIds is absent — leaf tasks have no children
}
Why plain objects? React state must be immutable — you never mutate an object in place, you replace it with a new copy. Plain objects make this easy: { ...oldTask, done: true } creates a new object with all old fields plus the changed one, leaving the original untouched. This is exactly like immutable data in functional programming.

// Returns today's date as a YYYY-MM-DD string in *local* time.
// Using local time is important: if you use new Date().toISOString() you get UTC,
// which can be off by a day depending on the user's timezone.
function todayStr() {
  const d = new Date();
  // padStart(2, "0") ensures single-digit months/days get a leading zero: "04" not "4"
  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}-${String(d.getDate()).padStart(2, "0")}`;
}

// A task is "active-deferred" when its deferUntil date is strictly after today.
// The string comparison works correctly for ISO dates: "2026-05-01" > "2026-04-30" is true.
// Using string comparison instead of Date objects avoids timezone parsing issues.
function isDeferred(task) {
  return !!(task.deferUntil && task.deferUntil > todayStr());
}

// Subtracts months/weeks from a YYYY-MM-DD base date.
// Used for the quick-pick buttons: "1 week before due", "1 month before due".
function subtractFromDate(base, { months = 0, weeks = 0 }) {
  const d = new Date(base + "T00:00:00"); // append T00:00:00 to force local-time parsing
  d.setMonth(d.getMonth() - months);
  d.setDate(d.getDate() - weeks * 7);
  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}-${String(d.getDate()).padStart(2, "0")}`;
}
effortToMinutes() + minutesToEffortLabel() — effort arithmetic App.jsx · lines 200–230 ↑ Effort Metadata 
// Converts a human-readable effort string to minutes.
// Uses substring matching rather than an exact lookup table, so user-defined
// values like "4.5 hours" or "3 days" work without any extra configuration.
function effortToMinutes(str) {
  if (!str) return 0;
  const s = str.toLowerCase().trim();
  const num = parseFloat(s); // extracts the leading number from strings like "2 hours"
  if (isNaN(num) || num <= 0) return 0;
  // Check from longest unit to shortest to avoid "month" matching "min"
  if (s.includes("month")) return Math.round(num * 9600);  // ~4 weeks × 5 days × 8 hours
  if (s.includes("week"))  return Math.round(num * 2400);  // 5 working days × 8 hours
  if (s.includes("day"))   return Math.round(num * 480);   // 8-hour working day
  if (s.includes("hour") || s.endsWith("h")) return Math.round(num * 60);
  if (s.includes("min")  || s.endsWith("m")) return Math.round(num);
  return 0;
}

// Converts minutes back to a compact human-readable label for display.
// Used in effort-total badges like "⏱ 3.5h" or "⏱ 2d".
function minutesToEffortLabel(minutes) {
  if (!minutes || minutes <= 0) return null;
  if (minutes < 60)   return `${minutes}m`;
  if (minutes < 480)  return `${+((minutes / 60).toFixed(1))}h`;   // 60–479 min → "Xh"
  if (minutes < 2400) return `${+((minutes / 480).toFixed(1))}d`;  // 480–2399 min → "Xd"
  if (minutes < 9600) return `${+((minutes / 2400).toFixed(1))}w`; // 2400–9599 → "Xw"
  return `${+((minutes / 9600).toFixed(1))}mo`;                    // 9600+ → "Xmo"
  // The unary + before ((...).toFixed(1)) removes trailing zeros: +(1.0) → 1, +(1.5) → 1.5
}
waterfallFilter() — Next Actions visibility rule App.jsx · lines 148–158 ↑ Waterfall & Group-By 
// GTD rule: only show the *next* action in a sequence, not all downstream steps.
// Level 2 tasks (direct children of a project) are always visible.
// Level 3+ tasks (children of next actions) only appear once their parent is done.
// This is a pure function — it only reads its inputs, has no side effects.
function waterfallFilter(nextTasks, allTasks) {
  return nextTasks.filter(task => {
    // Tasks with no parent (standalone actions) are always shown
    if (!task.parentId) return true;

    const parent = allTasks.find(t => t.id === task.parentId);
    if (!parent) return true; // orphaned task — show it rather than hiding it silently

    // Level 2: parent is a project root → always visible
    if (parent.bucket === "project") return true;

    // Level 3+: parent is another next action → only visible when parent is complete
    return !!parent.done;
  });
}
groupByField() — multi-mode task grouping App.jsx · lines 160–210 ↑ Waterfall & Group-By 
// Groups a flat task array by a named field.
// Returns an array of { key, label, items } objects — one per group.
// Multi-value fields (location, priority) use the first value as the group key.
function groupByField(taskList, field, allTasks = []) {
  // Each field type has a different label for tasks with no value set
  const ungroupedLabel = field === "project" ? "No Project"
                       : field === "effort"  ? "No Effort"
                       : "Ungrouped";
  const groups = {};   // { groupKey: [task, task, ...] }
  const ungrouped = []; // tasks with no value for this field

  taskList.forEach(task => {
    let keys = [];
    if      (field === "location") keys = task.location || [];
    else if (field === "priority") keys = task.priority || [];
    else if (field === "dueDate")  keys = task.dueDate ? [task.dueDate] : [];
    else if (field === "effort")   keys = task.effort  ? [task.effort]  : [];
    else if (field === "project") {
      // Walk the parentId chain all the way to the root project name
      if (task.parentId) {
        let cur = task;
        while (cur.parentId) {
          const parent = allTasks.find(t => t.id === cur.parentId);
          if (!parent) break;
          cur = parent;
        }
        if (cur.id !== task.id) keys = [cur.text]; // cur is now the root project
      }
    }

    if (!keys.length) { ungrouped.push(task); return; }
    const key = keys[0]; // multi-value fields: only the first value determines the group
    if (!groups[key]) groups[key] = [];
    groups[key].push(task);
  });

  // Sort groups: effort groups by duration (shortest first); all others alphabetically
  const sorted = Object.entries(groups)
    .sort(([a], [b]) => field === "effort"
      ? effortToMinutes(a) - effortToMinutes(b)  // numeric sort by duration
      : a.localeCompare(b))                        // alphabetic sort
    .map(([key, items]) => ({ key, label: key, items }));

  // Append the "no value" group at the bottom
  if (ungrouped.length) sorted.push({ key: "__ungrouped__", label: ungroupedLabel, items: ungrouped });
  return sorted;
}
countDescendants() — recursive task count with cycle guard App.jsx · lines 254–268 ↑ Descendant Count Badge 
// Recursively counts all descendants of a task via its childIds array.
// Returns { total, incomplete } so the badge can show "↓ 3 / 5".
// The visited Set prevents infinite loops if childIds ever forms a cycle
// due to a data corruption bug — standard depth-first graph traversal guard.
function countDescendants(taskId, allTasks, visited = new Set()) {
  if (visited.has(taskId)) return { total: 0, incomplete: 0 }; // cycle detected
  visited.add(taskId);

  const task = allTasks.find(t => t.id === taskId);
  if (!task) return { total: 0, incomplete: 0 };

  // Count this task itself
  const own = { total: 1, incomplete: task.done ? 0 : 1 };

  // Recursively count all children and accumulate
  const childTotals = (task.childIds || []).reduce(
    (acc, cid) => {
      const c = countDescendants(cid, allTasks, visited);
      return { total: acc.total + c.total, incomplete: acc.incomplete + c.incomplete };
    },
    { total: 0, incomplete: 0 } // starting accumulator
  );

  return {
    total:      own.total      + childTotals.total,
    incomplete: own.incomplete + childTotals.incomplete,
  };
}

sumDescendantEffort() — recursive effort roll-up App.jsx · lines 240–250 ↑ Collapsible Project Tree 
// Sums the effort of a task and ALL its descendants recursively.
// Replaces the old one-level reduce that only counted direct children.
// Returns minutes as an integer; caller converts with minutesToEffortLabel().
function sumDescendantEffort(taskId, allTasks, visited = new Set()) {
  if (visited.has(taskId)) return 0; // cycle guard — same pattern as countDescendants
  visited.add(taskId);

  const task = allTasks.find(t => t.id === taskId);
  if (!task) return 0;

  const own = effortToMinutes(task.effort); // this task's own effort in minutes

  // Sum recursively over all children
  const childTotal = (task.childIds || []).reduce(
    (sum, cid) => sum + sumDescendantEffort(cid, allTasks, visited),
    0
  );

  return own + childTotal;
}

// Parses the structured block that the AI appends to every project review response.
// Handles both arrow styles (→ U+2192 and ->) and both line-ending styles (\n and \r\n).
function extractSuggestions(text) {
  // Normalise: convert Windows line endings and ASCII arrows to a single canonical form
  const normalised = text.replace(/\r\n/g, "\n").replace(/->/g, "→");

  // Match everything after "→SUGGESTIONS:\n" to the end of the string
  const m = normalised.match(/→SUGGESTIONS:\n([\s\S]*)$/);
  if (!m) return []; // AI didn't include the block — return empty

  return m[1]
    .split("\n")
    // Strip leading "1. " or "- " numbering that the AI might add
    .map(l => l.replace(/^\d+\.\s*|-\s*/, "").trim())
    // Filter out empty lines and the "(none)" marker the AI uses for "no suggestions"
    .filter(l => l && l !== "(none)" && l !== "(none).");
}

// Parses the structured block from the "metadata quality" project review mode.
// Each line has the form: taskId|field:value|field:value
// Returns an array of { taskId, fields: { effort?, dueDate?, deferUntil? } }
function extractMetadata(text) {
  const normalised = text.replace(/\r\n/g, "\n").replace(/->/g, "→");
  const m = normalised.match(/→METADATA:\n([\s\S]*)$/);
  if (!m) return [];

  return m[1]
    .split("\n")
    .map(l => l.trim())
    // Only process lines that contain a | separator (the taskId|field:value format)
    .filter(l => l && l !== "(none)" && l !== "(none)." && l.includes("|"))
    .map(l => {
      // Split on | to get [taskId, "effort:2 hours", "due:2026-06-01", ...]
      const [taskId, ...pairs] = l.split("|");
      const fields = {};
      pairs.forEach(p => {
        const [k, v] = p.split(":").map(s => s.trim());
        if (k === "effort")  fields.effort     = v;
        if (k === "due")     fields.dueDate    = v;
        if (k === "defer")   fields.deferUntil = v;
      });
      return { taskId: taskId.trim(), fields };
    })
    // Only keep rows that have a valid taskId and at least one field suggestion
    .filter(r => r.taskId && Object.keys(r.fields).length > 0);
}

// parentId === null → root projects, ordered by their position in the flat tasks array.

  if (parentId === null) {

useResizer() — drag-to-resize custom hook App.jsx · lines 384–437 ↑ Resizable Panels 
// A custom hook packages reusable stateful logic.
// Convention: custom hooks must start with "use" so React knows they contain hooks.
// This hook manages a single resizable dimension (width or height).
//
// Parameters:
//   storageKey  — localStorage key to persist the size across sessions
//   defaultSize — initial size in pixels if nothing is stored yet
//   { min, max, direction, sign }
//     direction "h" = horizontal resize (width), "v" = vertical (height)
//     sign +1 = dragging right/down makes it bigger; -1 = opposite
function useResizer(storageKey, defaultSize, { min, max, direction = 'h', sign = 1 }) {
  // Read the persisted size from localStorage on first render only (lazy initializer)
  const [size, setSize] = useState(() => {
    const stored = localStorage.getItem(storageKey);
    return stored ? parseInt(stored, 10) : defaultSize;
  });

  // useRef gives us a mutable container that doesn't trigger re-renders.
  // We need this so the mousemove closure always reads the current size,
  // not the stale value captured when mousedown fired.
  const sizeRef = useRef(size);
  sizeRef.current = size; // keep ref in sync on every render

  // Save to localStorage whenever size changes
  useEffect(() => { localStorage.setItem(storageKey, size); }, [storageKey, size]);

  // handleMouseDown is the function we attach to the resize handle element.
  // It captures the starting position and size, then attaches temporary
  // mousemove/mouseup listeners to the *window* (not the handle element)
  // so the resize continues even if the mouse leaves the handle.
  const handleMouseDown = useCallback((e) => {
    e.preventDefault(); // prevent text selection during drag
    const startPos  = direction === 'h' ? e.clientX : e.clientY;
    const startSize = sizeRef.current;

    const onMove = (ev) => {
      const delta = sign * ((direction === 'h' ? ev.clientX : ev.clientY) - startPos);
      setSize(Math.min(max, Math.max(min, startSize + delta))); // clamp to [min, max]
    };

    const onUp = () => {
      window.removeEventListener('mousemove', onMove);
      window.removeEventListener('mouseup',   onUp);
      document.body.style.cursor     = ''; // restore cursor
      document.body.style.userSelect = ''; // restore text selection
    };

    document.body.style.cursor     = direction === 'h' ? 'col-resize' : 'row-resize';
    document.body.style.userSelect = 'none'; // prevent text selection while dragging
    window.addEventListener('mousemove', onMove);
    window.addEventListener('mouseup',   onUp);
  }, [min, max, direction, sign]);

  return [size, handleMouseDown]; // caller uses: const [width, onMouseDown] = useResizer(...)
}

// The visible resize handle — a thin 4px div that highlights on hover
function ResizeHandle({ onMouseDown, direction = 'h' }) {
  const [hovered, setHovered] = useState(false);
  const isH = direction === 'h';
  return (
    <div
      onMouseDown={onMouseDown}
      onMouseEnter={() => setHovered(true)}
      onMouseLeave={() => setHovered(false)}
      style={{
        flexShrink: 0,
        width:  isH ? 4 : '100%',
        height: isH ? '100%' : 4,
        cursor: isH ? 'col-resize' : 'row-resize',
        background: hovered ? COLORS.border2 : COLORS.border,
        transition: 'background 0.15s',
      }}
    />
  );
}
Why useRef for sizeRef? The onMove closure captures startSize at mousedown time. If we read from the React state variable inside onMove instead, we'd read a stale value (the state at the time mousedown ran). The ref is updated on every render, so it always reflects the current value — a safe bridge between React's render cycle and imperative event listeners.

// Called when the user changes the Project dropdown in TaskDetailPanel.
// Atomically: removes from old parent's childIds, updates task's parentId, adds to new parent's childIds.
const reassignProject = useCallback((taskId, newProjectId) => {
  setTasks(prev => {
    const task = prev.find(t => t.id === taskId);
    if (!task) return prev;
    const oldProjectId = task.parentId || null;
    if (oldProjectId === newProjectId) return prev; // no-op

    // Cycle guard: don't allow assigning to a descendant of the task itself
    if (newProjectId) {
      // isDescendant walks the childIds tree down from a given node
      function isDescendant(ancestorId, nodeId, seen = new Set()) {
        if (seen.has(nodeId) || !nodeId) return false;
        seen.add(nodeId);
        const node = prev.find(t => t.id === nodeId);
        return (node?.childIds || []).some(cid =>
          cid === ancestorId || isDescendant(ancestorId, cid, seen)
        );
      }
      if (isDescendant(taskId, newProjectId)) return prev;
    }

    // Apply all three changes in a single map pass
    return prev.map(t => {
      if (t.id === taskId)    return { ...t, parentId: newProjectId || null };
      // Remove from old parent's childIds
      if (oldProjectId && t.id === oldProjectId)
        return { ...t, childIds: (t.childIds || []).filter(id => id !== taskId) };
      // Add to new parent's childIds (guard against duplicates)
      if (newProjectId && t.id === newProjectId) {
        const existing = t.childIds || [];
        return existing.includes(taskId) ? t : { ...t, childIds: [...existing, taskId] };
      }
      return t;
    });
  });
}, []);

// Called wherever the app needs to colour a chip or dropdown to signal accuracy.
function effortAccuracyColor(estimatedMin, actualMin) {
  // If either value is missing, fall back to the neutral teal effort colour.
  if (!estimatedMin || !actualMin) return COLORS.effort;

  // delta = how much OVER estimate we went, as a fraction.
  //   0.0 = exactly on time, 0.25 = 25% over, -0.2 = 20% under
  const delta = (actualMin - estimatedMin) / estimatedMin;

  if (delta <= 0)    return "#5ab878"; // green  — on time or under
  if (delta <= 0.25) return "#d4a95a"; // amber  — up to 25% over
  return "#d45a5a";                    // red    — more than 25% over
}

// Builds a plain-text block appended to the AI system prompt
// only for "process" and "projectMetadata" modes.
// Returns "" when there is nothing useful to say.
function buildCalibrationContext(tasks, efforts, calibrationOverrides) {
  const completed = tasks.filter(t => t.done && t.effort && t.actualEffort);
  if (!completed.length && !Object.values(calibrationOverrides || {}).some(Boolean)) return "";

  // ── Per-label stats ──────────────────────────────────────
  const stats = {};
  efforts.forEach(label => { stats[label] = { totalActual: 0, count: 0 }; });
  completed.forEach(t => {
    if (stats[t.effort]) {
      stats[t.effort].totalActual += effortToMinutes(t.actualEffort);
      stats[t.effort].count       += 1;
    }
  });

  // ── Global average ratio ─────────────────────────────────
  const globalTotalActual = completed.reduce((s, t) => s + effortToMinutes(t.actualEffort), 0);
  const globalTotalEst    = completed.reduce((s, t) => s + effortToMinutes(t.effort), 0);
  const globalRatioPct    = globalTotalEst > 0
    ? Math.round(((globalTotalActual - globalTotalEst) / globalTotalEst) * 100)
    : null;

  // ── Build per-label lines ─────────────────────────────────
  const lines = [];
  efforts.forEach(label => {
    const override = calibrationOverrides?.[label];
    const s = stats[label];
    if (override) {
      lines.push(`- "${label}" → use "${override}" (manual override)`);
    } else if (s && s.count >= MIN_CALIBRATION_SAMPLES) {
      const avgMin   = Math.round(s.totalActual / s.count);
      const avgLabel = minutesToEffortLabel(avgMin) || label;
      const estMin   = effortToMinutes(label);
      const pct      = estMin ? Math.round(((avgMin - estMin) / estMin) * 100) : 0;
      lines.push(`- "${label}" → actual avg ~${avgLabel} (${pct > 0 ? "+" : ""}${pct}%, n=${s.count})`);
    } else if (s && s.count > 0) {
      lines.push(`- "${label}" → ${s.count} sample(s) so far (need ${MIN_CALIBRATION_SAMPLES})`);
    }
  });

  if (!lines.length && globalRatioPct === null) return "";

  let globalNote = "";
  if (globalRatioPct !== null && completed.length >= MIN_CALIBRATION_SAMPLES) {
    globalNote = globalRatioPct > 0
      ? `\nOverall this user underestimates by ~${globalRatioPct}% — suggest the next label up when unsure.`
      : `\nOverall this user estimates accurately (avg ${globalRatioPct}% variance).`;
  }

function extractUpdateAction(text) {
  // The 's' flag makes '.' match newlines too — needed for multi-line notes.
  const m = text.match(/→ACTION:update\|([^|\n]+)\|(.*)/s);
  if (!m) return null;

  const taskId   = m[1].trim();   // e.g. "lf8j2k5r"

  // 'notes' must be the last field because its value may contain '|'.
  // Find where 'notes:' starts (may be preceded by a pipe delimiter).
  const notesIdx   = fieldStr.search(/(^|\|)notes:/);
  const pureFields = notesIdx !== -1
    ? fieldStr.slice(0, notesIdx).replace(/\|$/, "")  // fields before notes
    : fieldStr;
  const notesRaw   = notesIdx !== -1
    ? fieldStr.slice(fieldStr.indexOf("notes:", notesIdx) + 6)  // value after "notes:"
    : null;

  // Build a changes object from the remaining pipe-delimited key:value pairs.
  const changes = {};
  pureFields.split("|").filter(Boolean).forEach(pair => {
    const colon = pair.indexOf(":");
    if (colon === -1) return;
    const key = pair.slice(0, colon).trim();
    const val = pair.slice(colon + 1).trim();
    if (key === "due")      changes.dueDate    = val;
    if (key === "defer")    changes.deferUntil = val;
    if (key === "effort")   changes.effort     = val;
    if (key === "bucket")   changes.bucket     = val;
    if (key === "title")    changes.text       = val;  // 'text' is the internal field name
    if (key === "priority") changes.priority   = val.split(",").map(s => s.trim()).filter(Boolean);
    if (key === "location") changes.location   = val.split(",").map(s => s.trim()).filter(Boolean);
  });

  // Decode \n back to real newlines in notes.
  // The AI is instructed to write \n; here we reverse that encoding.
  if (notesRaw !== null) changes.notes = notesRaw.replace(/\\n/g, "\n");

// Pure function — no React deps; can be called from any completion path.
function buildNextOccurrence(task) {
  const rec = task.recurrence;
  if (!rec) return null;
  // Anchor: either the original due date or today (completion date)
  const anchor = rec.rescheduleFrom === "dueDate" && task.dueDate
    ? task.dueDate : todayStr();
  const d = new Date(anchor + "T00:00:00");
  const { frequency, interval = 1, weekDays } = rec;

  if (frequency === "daily") {
    d.setDate(d.getDate() + interval);
  } else if (frequency === "weekly") {
    if (weekDays && weekDays.length > 0) {
      // Advance to next matching weekday (tries up to 14 days ahead)
      d.setDate(d.getDate() + 1);
      let tries = 0;
      while (!weekDays.includes(d.getDay()) && tries < 14) {
        d.setDate(d.getDate() + 1); tries++;
      }
    } else {
      d.setDate(d.getDate() + interval * 7);
    }
  } else if (frequency === "monthly") {
    d.setMonth(d.getMonth() + interval);
  } else if (frequency === "yearly") {
    d.setFullYear(d.getFullYear() + interval);
  }

  const nextDue = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,"0")}-${String(d.getDate()).padStart(2,"0")}`;
  // Strip completion-specific fields from the previous occurrence
  const { id: _id, done: _done, actualEffort: _ae, ...rest } = task;
  return {
    ...rest,
    id: genId(),
    done: false,
    dueDate: nextDue,
    created: Date.now(),

  };
}

async function doCalendarDeleteEvent(token, eventId) {
  const res = await fetch(
    `https://www.googleapis.com/calendar/v3/calendars/primary/events/${encodeURIComponent(eventId)}`,
    { method: 'DELETE', headers: { Authorization: `Bearer ${token}` } }
  );
  // 204 = success with no body; any other non-ok status is an error
  if (!res.ok && res.status !== 204) {
    const d = await res.json().catch(() => ({}));
    throw new Error(d.error?.message || `Calendar API ${res.status}`);
  }
}

// Patch (partially update) an event — only send the fields that are changing
async function doCalendarUpdateEvent(token, eventId, { summary, date, startTime, endTime }) {
  const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;
  const patch = {};                       // build only the fields we need
  if (summary) patch.summary = summary;
  if (date) {
    if (startTime && endTime) {           // timed event
      patch.start = { dateTime: `${date}T${startTime}:00`, timeZone: tz };
      patch.end   = { dateTime: `${date}T${endTime}:00`,   timeZone: tz };
    } else {                              // all-day event
      patch.start = { date };
      patch.end   = { date };
    }
  }
  const res = await fetch(
    `https://www.googleapis.com/calendar/v3/calendars/primary/events/${encodeURIComponent(eventId)}`,
    { method: 'PATCH',
      headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify(patch) }
  );
  if (!res.ok) { const d = await res.json(); throw new Error(d.error?.message || `Calendar API ${res.status}`); }
  return await res.json();  // returns the full updated event object
}
PUT vs PATCH. Google Calendar supports both PUT (replace the whole event) and PATCH (merge changes). PATCH is safer here because we only send the fields we want to change — title, date, times. A PUT would require sending every field of the existing event, including attendees and recurrence rules we haven't loaded. Always prefer PATCH when you're only changing part of a resource.

// Same pipe-delimited pattern as the task →ACTION: parsers.
// The AI ends its reply with exactly one of these lines:
//   →ACTION:calendar_create|Meeting title|date:2026-05-12|startTime:10:00|endTime:11:00
//   →ACTION:calendar_update|<event_id>|date:2026-05-14|startTime:09:00
//   →ACTION:calendar_delete|<event_id>

function extractCalendarCreateAction(text) {
  const m = text.match(/→ACTION:calendar_create\|([^|
]+)\|(.*)/s);
  if (!m) return null;
  const title = m[1].trim();
  const fields = {};
  m[2].split('|').filter(Boolean).forEach(pair => {
    const colon = pair.indexOf(':');
    if (colon === -1) return;
    const key = pair.slice(0, colon).trim();
    const val = pair.slice(colon + 1).trim();
    if (key === 'date')        fields.date        = val;
    if (key === 'startTime')   fields.startTime   = val;
    if (key === 'endTime')     fields.endTime     = val;
    if (key === 'description') fields.description = val;
    if (key === 'taskId')      fields.taskId      = val;  // links to existing task
  });
  if (!title || !fields.date) return null;
  return { title, ...fields };
}

function extractCalendarUpdateAction(text) {
  const m = text.match(/→ACTION:calendar_update\|([^|
]+)\|(.*)/s);
  if (!m) return null;
  const eventId = m[1].trim();   // first segment is the Google Calendar event ID
  const fields = {};
  m[2].split('|').filter(Boolean).forEach(pair => {
    const colon = pair.indexOf(':');
    if (colon === -1) return;
    const key = pair.slice(0, colon).trim();
    const val = pair.slice(colon + 1).trim();
    if (key === 'date')      fields.date      = val;
    if (key === 'startTime') fields.startTime = val;
    if (key === 'endTime')   fields.endTime   = val;
    if (key === 'title')     fields.title     = val;
    if (key === 'taskId')    fields.taskId    = val;
  });
  return eventId ? { eventId, ...fields } : null;
}

function extractCalendarDeleteAction(text) {
  const m = text.match(/→ACTION:calendar_delete\|([^
]+)/);
  return m ? { eventId: m[1].trim() } : null;
}

// Retry on 429/500/503 with Retry-After header support
async function fetchWithBackoff(url, options = {}, maxRetries = 3) {
  let attempt = 0;
  while (true) {
    const res = await fetch(url, options);
    if (res.ok) return res;
    const retryable = res.status === 429 || res.status === 500 || res.status === 503;
    if (!retryable || attempt >= maxRetries) return res;
    const retryAfter = res.headers.get('Retry-After');
    const waitMs = retryAfter ? parseFloat(retryAfter) * 1000 : (2 ** attempt) * 1000;
    await sleep(waitMs);
    attempt++;
  }
}

// Chunked Promise.all — fires at most batchSize requests concurrently
async function batchedAll(items, asyncFn, batchSize = 10, delayMs = 150) {
  const results = [];
  for (let i = 0; i < items.length; i += batchSize) {
    const chunk = items.slice(i, i + batchSize);
    const chunkResults = await Promise.all(chunk.map(asyncFn));
    results.push(...chunkResults);
    if (i + batchSize < items.length) await sleep(delayMs);
  }
  return results;
}
